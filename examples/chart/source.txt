#include <stdio.h>

#define MAX 4
int A[MAX], s1, s2;

void swap(int &a, int &b) {
  int t = a;
  a = b;
  b = t;
}

int main() {
  for (int i = 0; i < MAX; i++) {
	scanf("%d", &A[i]);
  }
  
  for (int i = 0; i < MAX - 1; i++) {
	for (int j = 0; j < MAX - 1 - i; j++) {
	  if (A[j] > A[j + 1]) {
		swap(A[j], A[j + 1]); s1 = j; s2 = j + 1;
	  }
	}
  }
  return 0;
}
